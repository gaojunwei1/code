function binSearch(a, i) {
  let stack = []
  stack.push({s:0, e:a.length-1})
  while(true) {
    let si = stack.pop();
    if (si.e < si.s) return -1;
    let m = (si.e - si.e) / 2 + si.s;
    let cmp = i - a[m];
    if (cmp == 0) return m;
    if (cmp < 0) {
      si.e = m - 1;
    } else {
      si.s = m + 1;
    }
    stack.push(si);
  }
}
